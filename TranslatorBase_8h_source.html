308 #define AL_USDMAYA_DEFINE_TRANSLATOR(PlugClass, TranslatedType) \
309 TfRefPtr<PlugClass> \
310 PlugClass::create(TranslatorContextPtr context) { \
311  TfType const &type = TfType::Find<TranslatedType>(); \
312  if(!type.IsUnknown()) { \
313  TfRefPtr<PlugClass> plugin = TfCreateRefPtr(new This()); \
314  plugin->setTranslatedType(type); \
315  plugin->setContext(context); \
316  if(!plugin->initialize()) return TfRefPtr<PlugClass>(); \
317  return plugin; \
318  } \
319  else { \
320  TF_CODING_ERROR( \
321  "Failed to get %s usd type, maybe the needed plugin is not loaded", \
322  typeid(TranslatedType).name()); \
323  return TfNullPtr; \
324  } \
325 } \
326  \
327 TF_REGISTRY_FUNCTION(TfType) \
328 { \
329  TfType::Define<PlugClass, TfType::Bases<TranslatorBase>>() \
330  .SetFactory<TranslatorFactory<PlugClass>>(); \
331 }
